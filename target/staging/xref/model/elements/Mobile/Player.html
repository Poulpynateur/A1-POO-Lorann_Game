13  public class Player extends Mobile{
14

17  
18      				/**** CONSTRUCTOR ****/
19      
20      /**
21       * Initiate the player.
22       * 
23       * @param stringStyle the stringStyle (example : "L1" corresponds to the player)
24       * @param x X coordinate
25       * @param y Y coordinate
26       */
27      private Player(String stringStyle, int x, int y) {
28          super(stringStyle,Types.PLAYER,x,y);
29      }
30  
31      public static synchronized void setInstance(String stringStyle, int x, int y)
32      {
33          if (INSTANCE == null)
34          {

36          }
37      }
38  
39      public static Player getInstance() {
40          return Player.INSTANCE;
41      }
42  
43      				/**** METHODS ****/
44      
45      /**
46       * Animate an element depends on the coordinates.
47       * 
48       * @param directionX Relative X coordinate
49       * @param directionY Relative Y coordinate
50       * 
51       * @see model.elements.ElementsList#getCharacter()
52       * - Get the character
53       *  @see model.elements.ElementsList#getImage()
54       * - Get the image
55       * @see model.elements.ElementsList#getDirection()
56       * - Get the direction
57       */
58      public void animate (int directionX, int directionY){
59  
60          if (directionX == 0 && directionY == 0) {
61              if (animationNumber < 7){
62                  this.sprite = ElementsList.values()[animationNumber].getImage();
63                  this.stringStyle = ElementsList.values()[animationNumber].getCharacter();
64                  animationNumber++;
65              }
66              else {
67                  this.sprite = ElementsList.values()[animationNumber].getImage();
68                  this.stringStyle = ElementsList.values()[animationNumber].getCharacter();
69                  animationNumber = 0;
70              }
71          }
72          else {
73              for (int i = 0; i < 8; i++) {
74                  if(ElementsList.values()[i].getDirection().x == directionX && ElementsList.values()[i].getDirection().y == directionY) {
75                      this.sprite = ElementsList.values()[i].getImage();
76                  }
77              }
78          }
79      }
80  }
