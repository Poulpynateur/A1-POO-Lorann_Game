20  public class AIDeplacement {
21  
22  	private IModel model;
23  	private List<Point> newEnemiesMove;
24  	private static final int DETECTION_RANGE=5;
25  
26  	//CONSTRUCTOR//
27  	public AIDeplacement(IModel model) {
28  		this.model = model;
29  		this.newEnemiesMove = new ArrayList<>();
30  	}
31  
32  	/**
33  	 * Generate new moves for monsters
34  	 * 
35  	 * @see  model.IModel#getEnemiesLocation()
36  	 * - Get enemies location
37  	 * @see model.IModel#getPlayerLocation()
38  	 * - Get player location
39  	 * @see  controller.GameManagement.AIDeplacement#getPath(Point enemyPos)
40  	 * - Get possible moves
41  	 * @return a list of Point : the move of each enemie
42  	 */
43  	public List<Point> moveAI() {
44  		newEnemiesMove = new ArrayList<>();
45   
46  		List<Point> enemiesPos = model.getEnemiesLocation();
47  		Random rand = new Random();
48  
49  		//Get the current player's position
50  		int random;
51  
52  		for (Point enemyPos : enemiesPos) {
53  
54  			List<Point> possiblePath = getPath(enemyPos);
55  
56  			if (possiblePath.size() <= 1) {
57  				random = 0;
58  			}
59  			else {
60  				random = rand.nextInt(possiblePath.size());
61  			}
62  
63  			if (isPlayerReachable(enemyPos,possiblePath) != null) {
64  				newEnemiesMove.add(isPlayerReachable(enemyPos,possiblePath));
65  			}
66  			else  {
67  				newEnemiesMove.add(new Point(possiblePath.get(random).x,possiblePath.get(random).y));
68  			}
69  		}
70  		return newEnemiesMove;
71  	}
72  
73  	/**
74  	 * Check if the player can be see and if he can be rush.
75  	 *
76  	 * @see AIDeplacement#isMovePossible(int, int, List)
77  	 *
78  	 * @param enemyPos position of the enemy to move.
79  	 * @param possiblePath list of possibles paths of this enemy
80  	 * @return an object of type Point : if case is reachable
81  	 */
82  	private Point isPlayerReachable(Point enemyPos,List<Point> possiblePath) {
83  		Point playerDirectionRelative = new Point(model.getPlayerLocation().x-enemyPos.x, model.getPlayerLocation().y-enemyPos.y);
84  		int moveX = playerDirectionRelative.x+enemyPos.x;
85  		int moveY = playerDirectionRelative.y+enemyPos.y;
86  
87  		if(Math.abs(playerDirectionRelative.x) < DETECTION_RANGE && Math.abs(playerDirectionRelative.y) < DETECTION_RANGE) {
88  			return isMovePossible(playerDirectionRelative.x,playerDirectionRelative.y, possiblePath);
89  		}
90  
91  		return null;
92  	}
93  
94  	/**
95  	 * Verify that the movement to the player is accessible (no obstacle)
96  	 * @param moveX Movement X to verify
97  	 * @param moveY Movement Y to verify
98  	 * @param possiblePath All possible path for this enemy
99  	 * @return null if no possible path found
100 	 */
101 	private Point isMovePossible(int moveX, int moveY, List<Point> possiblePath) {
102 		if(moveX < 0) {
103 			moveX = -1;
104 		}
105 		else if(moveX > 0) {
106 			moveX = 1;
107 		}
108 		if(moveY < 0) {
109 			moveY = -1;
110 		}
111 		else if(moveY > 0) {
112 			moveY = 1;
113 		}
114 		for (Point path : possiblePath) {
115 			if (path.x == moveX && path.y == moveY) {
116 				return path;
117 			}
118 		}
119 		return null;
120 	}
121 
122 	/**
123 	 * Get the different square around available
124 	 * @param enemyPos current enemies positions
125 	 * @return a list of Point : available moves
126 	 */
127 	private List<Point> getPath(Point enemyPos) {
128 
129 		List<Point> possiblePath = new ArrayList<>();
130 		Types typeCheck;
131 
132 		for (int x = -1; x <= 1; x++) {
133 			for (int y = -1; y <= 1; y++) {
134 				typeCheck = model.getType(enemyPos.x+x,enemyPos.y + y);
135 				if (typeCheck == Types.VOID || typeCheck == Types.PLAYER || typeCheck == Types.SPELL) {
136 					possiblePath.add(new Point(x,y));
137 				}
138 			}
139 		}
140 		return possiblePath;
141 	}
142 }
